function find(array, f, context) {
        var i,
            l;
        for (i = 0, l = array.length; i < l; i += 1) {
            if (array.hasOwnProperty(i) && f.call(context, array[i], i, array)) {
                return array[i];
            }
        }
        return null;
    }

function removeOne(array, item) {
        var found = array.indexOf(item);
        if (found > -1) {
            array.splice(found, 1);
        }
    }
