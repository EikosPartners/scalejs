function create (level) {
        return function() {
            if (!IE) {
                Array.prototype.unshift.call(arguments, level);
            } else {
                outstring = level + ' ';
                arguments.forEach(function(arg){
                    outstring += object.stringify(arg) + ' ';
                });
                arguments[0] = outstring;
            }

            log.apply(this, arguments);
        };
    }
